/ip/route
:foreach peer in=$peerBallowed do={

}
}
# end for A

:if ($selectedPeer = "B") do={
# add address
/ip address

/ip firewall filter

/ip/route
:foreach peer in=$peerAallowed do={
add dst-address=$peer gateway=$peerBname comment=$peerAname
}
